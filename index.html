        <p id="numbers">Press 'Generate' to get numbers</p>

        <button onclick="generateNumbers()">Generate</button>

        <button onclick="showAnswer()">Answer</button>

        <p id="answer" style="display:none;"></p>

    </div>

    

    <script>

        let num1, num2;

        

        function generateNumbers() {

            num1 = Math.floor(Math.random() * (100 - 5 + 1)) + 5;

            num2 = Math.floor(Math.random() * (100 - 5 + 1)) + 5;

            document.getElementById('numbers').innerText = `Numbers: ${num1} + ${num2}`;

            document.getElementById('answer').style.display = 'none';

        }

        

        function showAnswer() {

            if (num1 !== undefined && num2 !== undefined) {

                document.getElementById('answer').innerText = `Answer: ${num1 + num2}`;

                document.getElementById('answer').style.display = 'block';

            }

        }

    </script>
